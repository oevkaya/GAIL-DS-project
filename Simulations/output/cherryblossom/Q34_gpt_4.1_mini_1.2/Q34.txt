```python
import pandas as pd
import matplotlib.pyplot as plt

# Load the data from the CSV file
file_path = '/mnt/data/file-CkaqWNY14y7jXapdEbtxYT'
data = pd.read_csv(file_path)

# Filter to keep only runners of the "10 Mile" event
ten_mile_runners = data[data['event'] == '10 Mile']

# Extract the ages of the runners, dropping any missing values
ages = ten_mile_runners['age'].dropna()

# Plot histogram of ages
plt.figure(figsize=(10, 6))
plt.hist(ages, bins=15, color='skyblue', edgecolor='black')
plt.title('Age Distribution of 10 Mile Runners')
plt.xlabel('Age')
plt.ylabel('Frequency')
plt.grid(axis='y')
plt.show()

# Calculate summary statistics
summary_stats = {
    'mean_age': ages.mean(),
    'median_age': ages.median(),
    'std_dev_age': ages.std(),
    'min_age': ages.min(),
    'max_age': ages.max(),
    'count': ages.count()
}

print("Summary Statistics for the Ages of 10 Mile Runners:")
for stat, value in summary_stats.items():
    print(f"{stat}: {value:.2f}")
```
