```python
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.stats import mode

# Load the data
file_path = '/mnt/data/file-CkaqWNY14y7jXapdEbtxYT'
data = pd.read_csv(file_path)

# Filter for the "10 Mile" event
data_10m = data[data['event'] == '10 Mile']

# Extract ages and drop missing values
ages = data_10m['age'].dropna()

# Plot histogram
plt.figure(figsize=(10, 6))
sns.histplot(ages, bins=20, kde=False, color='blue')
plt.title('Age Distribution of Runners in the 10 Mile Event')
plt.xlabel('Age')
plt.ylabel('Number of Runners')
plt.grid(True)
plt.show()

# Calculate summary statistics
mean_age = ages.mean()
median_age = ages.median()
mode_age = mode(ages)[0][0]
std_age = ages.std()
min_age = ages.min()
max_age = ages.max()

print(f"Mean Age: {mean_age:.2f}")
print(f"Median Age: {median_age:.2f}")
print(f"Mode Age: {mode_age:.2f}")
print(f"Standard Deviation: {std_age:.2f}")
print(f"Minimum Age: {min_age}")
print(f"Maximum Age: {max_age}")
```

```python
import pandas as pd
import matplotlib.pyplot as plt

# Load the dataset
file_path = '/mnt/data/file-CkaqWNY14y7jXapdEbtxYT'
data = pd.read_csv(file_path)

# Filter for the "10 Mile" event
ten_mile_runners = data[data['event'] == '10 Mile']

# Extract ages of runners in the "10 Mile" event
ages = ten_mile_runners['age'].dropna()

# Plot histogram of ages
plt.figure(figsize=(10, 6))
plt.hist(ages, bins=20, color='skyblue', edgecolor='black')
plt.title('Age Distribution of Runners in the 10 Mile Event')
plt.xlabel('Age')
plt.ylabel('Number of Runners')
plt.grid(axis='y')
plt.show()

# Calculate summary statistics
mean_age = ages.mean()
median_age = ages.median()
std_age = ages.std()
min_age = ages.min()
max_age = ages.max()

# Print summary statistics
print(f'Mean Age: {mean_age:.2f}')
print(f'Median Age: {median_age:.2f}')
print(f'Standard Deviation: {std_age:.2f}')
print(f'Minimum Age: {min_age}')
print(f'Maximum Age: {max_age}')
```
